28 /// @brief A simple synchronization lock. Only one thread at a time can Acquire.
29 class Mutex {
30  public:
31  /// @enum Mode
32  ///
33  /// @brief Bitfield that describes the mutex configuration.
34  /// **Enumerations**:
35  ///
36  /// * `kModeNonRecursive` (`0`) - The mutex is initialized as a non-recursive
37  /// mutex.
38  /// * `kModeRecursive` (`1`) - The mutex is initialized as a recursive mutex.
39  enum Mode {
40  kModeNonRecursive = (0 << 0),
41  kModeRecursive = (1 << 0),
42  };
43 
44  /// @brief Default constructor that initializes a mutex as a recursive one.
45  Mutex() { Initialize(kModeRecursive); }
46 
47  /// @brief Constructor that initializes a mutex with a parameter.
48  /// @param[in] mode Mode indicating the mutex's recursive setting.
49  explicit Mutex(Mode mode) { Initialize(mode); }
50 
51  ~Mutex() {
52 #if !defined(_WIN32)
53  int ret = pthread_mutex_destroy(&mutex_);
54  assert(ret == 0);
55  (void)ret;
56 #else
57  CloseHandle(synchronization_object_);
58 #endif // !defined(_WIN32)
59  }
60 
61  /// @brief Acquire the mutex's ownership.
62  void Acquire() {
63 #if !defined(_WIN32)
64  int ret = pthread_mutex_lock(&mutex_);
65  assert(ret == 0);
66  (void)ret;
67 #else
68  WaitForSingleObject(synchronization_object_, INFINITE);
69 #endif // !defined(_WIN32)
70  }
71 
72  /// @brief Try to acquire the mutex's ownership.
73  bool TryLock() {
74 #if !defined(_WIN32)
75  int ret = pthread_mutex_trylock(&mutex_);
76  return !ret;
77 #else
78  auto ret = WaitForSingleObject(synchronization_object_, 0);
79  return ret == WAIT_OBJECT_0;
80 #endif // !defined(_WIN32)
81  }
82 
83  /// @brief Release the mutex's ownership.
84  void Release() {
85 #if !defined(_WIN32)
86  int ret = pthread_mutex_unlock(&mutex_);
87  assert(ret == 0);
88  (void)ret;
89 #else
90  if (mode_ & kModeRecursive) {
91  ReleaseMutex(synchronization_object_);
92  } else {
93  ReleaseSemaphore(synchronization_object_, 1, 0);
94  }
95 #endif // !defined(_WIN32)
96  }
97 
98  private:
99  void Initialize(Mode mode) {
100 #if !defined(_WIN32)
101  pthread_mutexattr_t attr;
102  int ret = pthread_mutexattr_init(&attr);
103  assert(ret == 0);
104  if (mode & kModeRecursive) {
105  ret = pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
106  assert(ret == 0);
107  }
108  ret = pthread_mutex_init(&mutex_, &attr);
109  assert(ret == 0);
110  ret = pthread_mutexattr_destroy(&attr);
111  assert(ret == 0);
112  (void)ret;
113 #else
114  mode_ = mode;
115  if (mode & kModeRecursive) {
116  synchronization_object_ = CreateMutex(nullptr, FALSE, nullptr);
117  } else {
118  synchronization_object_ = CreateSemaphore(nullptr, 1, 1, nullptr);
119  }
120 #endif // !defined(_WIN32)
121  }
122 
123 #if !defined(_WIN32)
124  pthread_mutex_t mutex_;
125 #else
126  HANDLE synchronization_object_;
127  Mode mode_;
128 #endif // !defined(_WIN32)
129 };

131 /// @brief Acquire and hold a /ref Mutex, while in scope.
132 ///
133 /// Example usage:
134 /// \code{.cpp}
135 /// Mutex syncronization_mutex;
136 /// void MyFunctionThatRequiresSynchronization() {
137 /// MutexLock lock(syncronization_mutex);
138 /// // ... logic ...
139 /// }
140 /// \endcode
141 class MutexLock {
142  public:
143  /// @brief Acquires specified mutex's ownership for a life time of the object.
144  ///
145  /// @param[in] mutex Mutex to aquire an ownership.
146  explicit MutexLock(Mutex& mutex) : mutex_(&mutex) { mutex_->Acquire(); }
147  ~MutexLock() { mutex_->Release(); }
148 
149  private:
150  // Copy is disallowed.
151  MutexLock(const MutexLock& rhs);
152  MutexLock& operator=(const MutexLock& rhs);
153 
154  Mutex* mutex_;
155 };

157 /// @brief Acquires and hold a /ref Mutex, if not held by someone else.
158 ///
159 /// Example usage:
160 /// \code{.cpp}
161 /// Mutex syncronization_mutex;
162 /// bool MyFunctionThatRequiresSynchronization() {
163 /// MutexTryLock lock;
164 /// if (!lock.Try(syncronization_mutex)) return false;
165 /// // ... logic ...
166 /// return true;
167 /// }
168 /// \endcode
169 class MutexTryLock {
170  public:
171  /// @brief Acquires specified mutex's ownership for a life time of the object.
172  ///
173  /// @param[in] mutex Mutex to aquire an ownership.
174  explicit MutexTryLock() : mutex_(nullptr) {}
175  ~MutexTryLock() { if (mutex_) mutex_->Release(); }
176 
177  /// @brief Acuires specified mutex's ownership for a life time of the object.
178  bool Try(Mutex& mutex) {
179  assert(mutex_ == nullptr);
180  const bool locked = mutex.TryLock();
181  if (locked) {
182  mutex_ = &mutex;
183  }
184  return locked;
185  }
186 
187  private:
188  // Copy is disallowed.
189  MutexTryLock(const MutexTryLock& rhs);
190  MutexTryLock& operator=(const MutexTryLock& rhs);
191 
192  Mutex* mutex_;
193 };
