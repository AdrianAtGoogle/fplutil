32 /// @class IndexAllocator "fplutil/index_allocator.h"
33 /// @brief Allocate, free, and defragment array indices.
34 ///
35 /// Purpose
36 /// =======
37 /// Allocate and free indices into an array. Tries to keep the array as small
38 /// as possible by recycling indices that have been freed.
39 ///
40 /// Example Usage
41 /// =============
42 /// We have an array of items that we would like to process with SIMD
43 /// instructions. Items can be added and deleted from the array though. We don't
44 /// want many unused indices in the array, since these holes still have to be
45 /// processed with SIMD (which processes indices in groups of 4 or 8 or 16).
46 ///
47 /// The IndexAllocator is great for this situation since you can call
48 /// Defragment() before running the SIMD algorithm. The Defragment() call will
49 /// backfill unused indices and ensure the data is contiguous.
50 ///
51 /// Details
52 /// =======
53 /// Periodically, you can call Defragment() to backfill indices that have been
54 /// freed with the largest indices. This minimizes the length of the array, and
55 /// more importantly makes the array data contiguous.
56 ///
57 /// During Defragment() when an index is moved, a callback
58 /// CallbackInterface::MoveIndex() is called so that the user can move the
59 /// corresponding data.
60 ///
61 /// Whenever the array size is increased (durring Alloc()) or decreased (during
62 /// Defragment()), a callback CallbackInterface::SetNumIndices() is called so
63 /// that the user can grow or shrink the corresponding data.
64 template <class Index, class Count>
65 class IndexAllocator {
66  public:
67  class CallbackInterface {
68  public:
69  virtual ~CallbackInterface() {}
70  virtual void SetNumIndices(Index num_indices) = 0;
71  virtual void MoveIndex(Index old_index, Index new_index) = 0;
72  };

74  /// Create an empty IndexAllocator that uses the specified callback
75  /// interface.
76  explicit IndexAllocator(CallbackInterface& callbacks)
77  : callbacks_(&callbacks) {
78  static_assert(std::is_signed<Count>::value, "Count must be a signed type");
79  }

81  /// If a previously-freed index can be recycled, allocates that index.
82  /// Otherwise, increases the total number of indices by `count`, and return
83  /// the first new index. When the number of indices is increased,
84  /// the SetNumIndices() callback is called.
85  /// @param count The number of indices in this allocation. Each block of
86  /// allocated indices is kept contiguous during Defragment()
87  /// calls. The index returned is the first index in the block.
88  Index Alloc(Count count) {
89  // Recycle an unused index, if one exists and has the correct count.
90  typename std::vector<Index>::iterator least_excess_it = unused_indices_.end();
91  Count least_excess = std::numeric_limits<Count>::max();
92  for (auto it = unused_indices_.begin(); it != unused_indices_.end(); ++it) {
93  const Index unused_index = *it;
94  const Count excess = CountForIndex(unused_index) - count;
95 
96  // Not big enough.
97  if (excess < 0)
98  continue;
99 
100  // Perfect size. Remove from `unused_indices_` pool.
101  if (excess == 0) {
102  unused_indices_.erase(it);
103  return unused_index;
104  }
105 
106  // Too big. We'll return the one with the least excess size.
107  if (excess < least_excess) {
108  least_excess = excess;
109  least_excess_it = it;
110  }
111  }
112 
113  // The unused index has a count that's too high.
114  if (least_excess_it != unused_indices_.end()) {
115  // Return the first `count` indices.
116  const Index excess_index = *least_excess_it;
117 
118  // Put the remainder in the `unused_indices_` pool.
119  const Index remainder_index = excess_index + count;
120  InitializeIndex(remainder_index, least_excess);
121  *least_excess_it = remainder_index;
122 
123  return excess_index;
124  }
125 
126  // Allocate a new index.
127  const Index new_index = num_indices();
128  SetNumIndices(new_index + count);
129  InitializeIndex(new_index, count);
130  return new_index;
131  }
132 
133  /// Recycle 'index'. It will be used in the next allocation, or backfilled in
134  /// the next call to Defragment().
135  /// @param index Index to be freed. Must be in the range
136  /// [0, num_indices_ - 1].
137  void Free(Index index) {
138  assert(ValidIndex(index));
139  unused_indices_.push_back(index);
140  }
141 
142  /// Backfill all unused indices with the largest indices by calling
143  /// callbacks_->MoveIndex(). This reduces the total number of indices,
144  /// and keeps memory contiguous. Contiguous memory is important to mimimize
145  /// cache misses.
146  ///
147  /// Note that we could eliminate Defragment() function by calling MoveIndex()
148  /// from Free(). The code would be simpler. We move the indices lazily,
149  /// however, for performance: Defragment() is something that can happen on a
150  /// background thread.
151  ///
152  /// This function is fairly cheap. If there are N holes, then there will be
153  /// **at most** N calls to MoveIndex(). We assume that moving an index is
154  /// cheaper than processing data for an index. So, you should Defragment()
155  /// right before you process data, for optimal performance.
156  ///
157  /// Note that the number of indices shrinks or stays the same in this
158  /// function, so the final call to SetNumIndices() will never result in a
159  /// reallocation of the underlying array (which would be slow).
160  ///
161  void Defragment() {
162  // Quick check is an optimization.
163  if (unused_indices_.size() == 0) return;
164 
165  // We check if unused index is the last index, so must be in sorted order.
166  std::sort(unused_indices_.begin(), unused_indices_.end());
167 
168  // Plug every unused index.
169  Index new_num_indices = num_indices();
170  while (unused_indices_.size() > 0) {
171 
172  // Recycle the largest unused undex.
173  const Index unused_index = unused_indices_.back();
174  unused_indices_.pop_back();
175 
176  // Search from the back for an index that fills the hole.
177  const Count count = CountForIndex(unused_index);
178  const Index fill_index = LastIndexMatchingCount(unused_index);
179 
180  // Only perform move if source and destination are different.
181  if (fill_index != unused_index) {
182  // Move fill element into unused index.
183  callbacks_->MoveIndex(fill_index, unused_index);
184  }
185 
186  // Shift all items after fill_index forward, to fill the new hole.
187  // The hope is that this won't have to move very many, since this is slow.
188  // TODO OPT: We can do better than this by finding indices of smaller
189  // size and using those to fill the hole. The assumption is
190  // that most of the time the allocation sizes will be pretty
191  // uniform, so this function will almost always be a no-op.
192  // But in pathological cases, this assumption can be very wrong.
193  MoveAllLaterIndices(fill_index);
194 
195  // Remember how many indices we've defragged.
196  new_num_indices -= count;
197  }
198 
199  // All unused indices have been filled in.
200  unused_indices_.clear();
201 
202  // The index array has shrunk. Notify with a callback. Note that since
203  // we're shrinking the array, this will not have to result in a realloc.
204  SetNumIndices(new_num_indices);
205  }

207  /// Returns true if there are no indices allocated.
208  bool Empty() const {
209  return num_indices() == NumUnusedIndices();
210  }

212  /// Returns true if the index is current allocated.
213  bool ValidIndex(Index index) const {
214  if (index < 0 || index >= num_indices())
215  return false;
216 
217  if (counts_[index] == 0)
218  return false;
219 
220  for (ConstIndexIterator it = unused_indices_.begin();
221  it != unused_indices_.end(); ++it) {
222  if (index == *it) return false;
223  }
224 
225  return true;
226  }

228  /// Returns the number of wasted indices. These holes will be plugged when
229  /// Degragment() is called.
230  Index NumUnusedIndices() const {
231  Count count = 0;
232  for (size_t i = 0; i < unused_indices_.size(); ++i) {
233  count += CountForIndex(unused_indices_[i]);
234  }
235  return count;
236  }

238  /// Returns the `count` value specified in Alloc. That is, the number of
239  /// consecutive indices associated with `index`.
240  Count CountForIndex(Index index) const {
241  assert(counts_[index] > 0);
242  return counts_[index];
243  }

245  /// Returns the size of the array that number of contiguous indices.
246  /// This includes all the indices that have been free.
247  Index num_indices() const { return static_cast<Index>(counts_.size()); }

253  /// Returns the next allocated index. Skips over all indices associated
254  /// with `index`.
255  Index NextIndex(Index index) const {
256  assert(index < num_indices() && counts_[index] > 0);
257  return index + counts_[index];
258  }
259 
260  /// Returns the previous allocated index. Skips over all indices associated
261  /// with `index` - 1.
262  Index PrevIndex(Index index) const {
263  assert(0 < index && index <= num_indices() &&
264  (index == num_indices() ||
265  counts_[index - 1] == 1 || counts_[index - 1] < 0));
266  const Count prev_count = counts_[index - 1];
267  return prev_count > 0 ? index - 1 : index - 1 + prev_count;
268  }

270  /// Set up the `counts_` array to hold the size of `index`. Only the value
271  /// at `counts_[index]` really matters. The others are initialized for
272  /// debugging, and to make traversal of the `counts_` array easier.
273  void InitializeIndex(Index index, Count count) {
274  // Initialize the count for this index.
275  counts_[index] = count;
276  for (int i = 1; i < count; ++i) {
277  counts_[index + i] = -i;
278  }
279  }
280 
281  /// Adjust internal state to match the new index size, and notify the
282  /// callback that size has changed.
283  void SetNumIndices(Index new_num_indices) {
284  // Increase (or decrease) the count logger.
285  counts_.resize(new_num_indices, 0);
286 
287  // Report size change.
288  callbacks_->SetNumIndices(new_num_indices);
289  }
290 
291  /// Returns allocation closest to the end of the array that has the matching
292  /// size of `search_index`.
293  Index LastIndexMatchingCount(Index search_index) const {
294  const Count count = CountForIndex(search_index);
295  Index index = num_indices();
296  for (;;) {
297  index = PrevIndex(index);
298  if (index == search_index)
299  break;
300 
301  if (CountForIndex(index) == count)
302  break;
303  }
304  assert(ValidIndex(index));
305  return index;
306  }
307 
308  /// Shift all indices after `index` to the left to fill the space occupied
309  /// by `index`.
310  void MoveAllLaterIndices(Index index) {
311  // Notify callback.
312  const Index next_index = NextIndex(index);
313  const Index num_deleted_indices = next_index - index;
314  for (Index i = next_index; i < num_indices(); i = NextIndex(i)) {
315  callbacks_->MoveIndex(i, i - num_deleted_indices);
316  }
317 
318  // Update counts_ to fill hole.
319  std::memmove(&counts_[index], &counts_[next_index], counts_.size() - next_index);
320  }
321 
322  // When indices are moved or the number of inidices changes, we notify the
323  // caller via these callbacks.
324  CallbackInterface* callbacks_;
325 
326  // For every valid index, the number of indices
327  // associated with that index. For intermediate indices, negative number
328  // representing the offset to the actual index.
329  //
330  // valid indices
331  // | | | | |
332  // v v v v v
333  // For example: 1 | 2 -1 | 4 -1 -2 -3 | 1 | 1
334  // ^ ^ ^ ^
335  // | | | |
336  // offset to the actual index
337  std::vector<Count> counts_;
338 
339  // When an index is freed, we keep track of it here. When an index is
340  // allocated, we use one off this array, if one exists.
341  // When Defragment() is called, we empty this array by filling all the
342  // unused indices with the highest allocated indices. This reduces the total
343  // size of the data arrays.
344  std::vector<Index> unused_indices_;
345 };
