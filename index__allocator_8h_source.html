37 /// @class IndexAllocator "fplutil/index_allocator.h"
38 /// @brief Allocate, free, and defragment array indices.
39 ///
40 /// Purpose
41 /// =======
42 /// Allocate and free indices into an array. Tries to keep the array as small
43 /// as possible by recycling indices that have been freed.
44 ///
45 /// Example Usage
46 /// =============
47 /// We have an array of items that we would like to process with SIMD
48 /// instructions. Items can be added and deleted from the array though. We don't
49 /// want many unused indices in the array, since these holes still have to be
50 /// processed with SIMD (which processes indices in groups of 4 or 8 or 16).
51 ///
52 /// The IndexAllocator is great for this situation since you can call
53 /// Defragment() before running the SIMD algorithm. The Defragment() call will
54 /// backfill unused indices and ensure the data is contiguous.
55 ///
56 /// Details
57 /// =======
58 /// Periodically, you can call Defragment() to backfill indices that have been
59 /// freed with the largest indices. This minimizes the length of the array, and
60 /// more importantly makes the array data contiguous.
61 ///
62 /// During Defragment() when an index is moved, a callback
63 /// CallbackInterface::MoveIndex() is called so that the user can move the
64 /// corresponding data.
65 ///
66 /// Whenever the array size is increased (durring Alloc()) or decreased (during
67 /// Defragment()), a callback CallbackInterface::SetNumIndices() is called so
68 /// that the user can grow or shrink the corresponding data.
69 template <class Index>
70 class IndexAllocator {
71  public:

73  /// `counts_` array can be as long as the largest index.

105  /// If a previously-freed index can be recycled, allocates that index.
106  /// Otherwise, increases the total number of indices by `count`, and return
107  /// the first new index. When the number of indices is increased,
108  /// the SetNumIndices() callback is called.
109  /// @param count The number of indices in this allocation. Each block of
110  /// allocated indices is kept contiguous during Defragment()
111  /// calls. The index returned is the first index in the block.
112  Index Alloc(Count count) {
113  // Recycle an unused index, if one exists and has the correct count.
114  typename std::vector<Index>::iterator least_excess_it = unused_indices_.end();
115  Count least_excess = std::numeric_limits<Count>::max();
116  for (auto it = unused_indices_.begin(); it != unused_indices_.end(); ++it) {
117  const Index unused_index = *it;
118  const Count excess = CountForIndex(unused_index) - count;
119 
120  // Not big enough.
121  if (excess < 0)
122  continue;
123 
124  // Perfect size. Remove from `unused_indices_` pool.
125  if (excess == 0) {
126  unused_indices_.erase(it);
127  return unused_index;
128  }
129 
130  // Too big. We'll return the one with the least excess size.
131  if (excess < least_excess) {
132  least_excess = excess;
133  least_excess_it = it;
134  }
135  }
136 
137  // The unused index has a count that's too high.
138  if (least_excess_it != unused_indices_.end()) {
139  // Return the first `count` indices.
140  const Index excess_index = *least_excess_it;
141  InitializeIndex(excess_index, count);
142 
143  // Put the remainder in the `unused_indices_` pool.
144  const Index remainder_index = excess_index + count;
145  InitializeIndex(remainder_index, least_excess);
146  *least_excess_it = remainder_index;
147 
148  return excess_index;
149  }
150 
151  // Allocate a new index.
152  const Index new_index = num_indices();
153  SetNumIndices(new_index + count);
154  InitializeIndex(new_index, count);
155  return new_index;
156  }

158  /// Recycle 'index'. It will be used in the next allocation, or backfilled in
159  /// the next call to Defragment().
160  /// @param index Index to be freed. Must be in the range
161  /// [0, num_indices_ - 1].
162  void Free(Index index) {
163  assert(ValidIndex(index));
164  unused_indices_.push_back(index);

167  // Only one block of unused indices left, and they're at the end of the
168  // array.
169  bool UnusedAtEnd() const {
170  return unused_indices_.size() == 1 &&
171  NextIndex(unused_indices_[0]) == num_indices();
172  }

174  /// Backfill all unused index blocks. That is, move index blocks around
175  /// until all the unused index blocks have the *highest* indices. Then,
176  /// shrink the number of indices to remove all unused index blocks.
177  ///
178  /// Every time we move an index block, we call callbacks_->MoveIndexRange().
179  /// In MoveIndexRange(), the callee can correspondingly move its
180  /// internal data around to match the index shuffle. At the end of
181  /// Degragment(), the callee's internal data will be contiguous.
182  /// Contiguous data is essential in data-oriented design, since it
183  /// minimizes cache misses.
184  ///
185  /// Note that we could eliminate Defragment() function by calling MoveIndex()
186  /// from Free(). The code would be simpler. We move the indices lazily,
187  /// however, for performance: Defragment() is something that can happen on a
188  /// background thread.
189  ///
190  /// This function has worst case runtime of O(n) index moves, where n is
191  /// the total number of indices. To see why, notice that indices are only
192  /// moved forward, and are always moved into the forward-most hole.
193  ///
194  /// Note that there is some inefficiency with setting the `count_` array
195  /// excessively. The worst-case number of operations on the `count_` array
196  /// is greater than O(n). However, the assumption is that since `count_` is
197  /// just an array of integers, operations on it are insignificant compared
198  /// to the actual data movement that happens in callbacks_->MoveIndexRange().
199  /// However, there is an optimization opportunity here, most likely.
200  ///
201  /// In practice, this function will normally perform much better than O(n)
202  /// moves. We endeavour to fill holes with index blocks near the end of the
203  /// array. That is, we try to leapfrog the hole to the end of the array
204  /// when possible.
205  ///
206  /// Because of this, when all allocations are the same size, the worst case
207  /// runtime improves significantly to O(k) index moves, where k is the
208  /// total number of *unused* indices.
209  ///
210  /// If moving an index is cheaper than processing data for an index,
211  /// then you should call Defragment() right before you process data,
212  /// for optimal performance.
213  ///
214  /// Note that the number of indices shrinks or stays the same in this
215  /// function, so the final call to SetNumIndices() will never result in a
216  /// reallocation of the underlying array (which would be slow).
217  ///
218  void Defragment() {
219  // Quick check. An optimization.
220  if (unused_indices_.size() == 0) return;
221 
222  for (;;) {
223  // We check if unused index is the last index, so must be in sorted order.
224  ConsolidateUnusedIndices();
225 
226  // If all the holes have been pushed to the end, we are done and can
227  // trim the number of indices.
228  const bool unused_at_end = unused_indices_.size() == 1 &&
229  NextIndex(unused_indices_[0]) == num_indices();
230  if (unused_at_end) break;
231 
232  // Find range of indices that will fit into the first block of
233  // unused indices and move them into it.
234  BackfillFirstUnused();
235  }
236 
237  // Remove hole at end.
238  SetNumIndices(unused_indices_[0]);
239  unused_indices_.clear();
240  }

247  /// Returns true if the index is current allocated.
248  bool ValidIndex(Index index) const {
249  if (index < 0 || index >= num_indices())
250  return false;
251 
252  if (counts_[index] == 0)
253  return false;
254 
255  for (ConstIndexIterator it = unused_indices_.begin();
256  it != unused_indices_.end(); ++it) {
257  if (index == *it) return false;
258  }
259 
260  return true;
261  }

280  /// Assert if the internal state is invalid in any way.
281  /// Sanity check on this data structure.
282  void VerifyInternalState() const {
283 #if FPL_INDEX_ALLOCATOR_VERIFY_INTERNAL_STATE
284  for (Index i = 0; i < num_indices();) {
285  // Each block of indices must start with the positive size of the block.
286  const Count count = counts_[i];
287  assert(count > 0);
288 
289  // Succeeding elements in a block give the offset back to the start.
290  for (Index j = 1; j < count; ++j) {
291  assert(counts_[i + j] == -j);
292  }
293 
294  // Jump to the next block.
295  i += count;
296  }
297 #endif // FPL_INDEX_ALLOCATOR_VERIFY_INTERNAL_STATE

308  /// Returns the next allocated index. Skips over all indices associated
309  /// with `index`.
310  Index NextIndex(Index index) const {
311  assert(0 <= index && index < num_indices() && counts_[index] > 0);
312  return index + counts_[index];
313  }

315  /// Returns the previous allocated index. Skips over all indices associated
316  /// with `index` - 1.
317  Index PrevIndex(Index index) const {
318  assert(0 < index && index <= num_indices() &&
319  (index == num_indices() ||
320  counts_[index - 1] == 1 || counts_[index - 1] < 0));
321  const Count prev_count = counts_[index - 1];
322  return prev_count > 0 ? index - 1 : index - 1 + prev_count;
323  }

325  /// Set up the `counts_` array to hold the size of `index`. Only the value
326  /// at `counts_[index]` really matters. The others are initialized for
327  /// debugging, and to make traversal of the `counts_` array easier.
328  void InitializeIndex(Index index, Count count) {
329  // Initialize the count for this index.
330  counts_[index] = count;
331  for (Count i = 1; i < count; ++i) {
332  counts_[index + i] = -i;
333  }
334  }

346  /// Combine adjacent blocks of unused incides in `unused_indices_`.
347  void ConsolidateUnusedIndices() {
348  // First put the indices in order so that we can process them efficiently.
349  std::sort(unused_indices_.begin(), unused_indices_.end());
350 
351  // Consolidate adjacent blocks of unused indices.
352  size_t new_num_unused = 0;
353  for (size_t i = 0; i < unused_indices_.size();) {
354  const Index unused = unused_indices_[i];
355 
356  // Find first non-consecutive index in unused_indices_.
357  size_t j = i + 1;
358  while (j < unused_indices_.size() &&
359  unused_indices_[j] == NextIndex(unused_indices_[j - 1])) {
360  ++j;
361  }
362 
363  // Consolidate consecutive unused indices.
364  const size_t num_consecutive = j - i;
365  if (num_consecutive > 1) {
366  const Count consolidated_count =
367  NextIndex(unused_indices_[j - 1]) - unused;
368  InitializeIndex(unused, consolidated_count);
369  }
370 
371  // Write to the output array.
372  unused_indices_[new_num_unused++] = unused;
373 
374  // Increment the read-counter, skipping over any we've consolidated.
375  i += num_consecutive;
376  }
377 
378  // Shrink number of unused indices. Size can only get smaller.
379  assert(new_num_unused <= unused_indices_.size());
380  unused_indices_.resize(new_num_unused);
381  }

383  /// Move later blocks of indices into the first hole in `unused_indices_`.
384  /// That is, move the first hole farther back in the index array.
385  void BackfillFirstUnused() {
386  assert(unused_indices_.size() > 0);
387  const IndexRange unused_range(
388  unused_indices_[0],
389  unused_indices_[0] + CountForIndex(unused_indices_[0]));
390 
391  // Find a fill_range after unused_range that we can move into unused_range.
392  //
393  // Case 1. Fill
394  // indices in fill_range are moved into the first part of unused_range.
395  // fill_range.Length() <= unused_range.Length()
396  //
397  // before: unused_range fill_range
398  // |..................| |abcdefghij|
399  //
400  // after: |abcdefghij|.......| |..........|
401  // unused_hole fill_hole
402  //
403  //
404  // Case 2. Shift
405  // indices in fill_range are moved to the left, into unused_range.
406  // this opens up a hole where fill_range used to end
407  //
408  // before: unused_range fill_range
409  // |............|abcdefghijklmnop|
410  //
411  // after: |abcdefghijklmnop|............|
412  // shift_hole
413  //
414  IndexRange fill_range = LastIndexRangeSmallerThanHole(unused_range.start());
415  const bool is_fill = fill_range.Valid();
416  if (!is_fill) {
417  // If there's no index range that will fit into the hole, shift over
418  // all the indices between this hole and the next.
419  const Index next_hole_index =
420  unused_indices_.size() > 1 ? unused_indices_[1] : num_indices();
421  fill_range = IndexRange(NextIndex(unused_range.start()), next_hole_index);
422  }
423 
424  // Allow the callback to move data associated with the indices.
425  callbacks_->MoveIndexRange(fill_range, unused_range.start());
426 
427  // Move `counts_` to fill_range's new location
428  memmove(&counts_[unused_range.start()], &counts_[fill_range.start()],
429  fill_range.Length() * sizeof(counts_[0]));
430 
431  // Re-initialize `counts_` for new holes.
432  if (is_fill) {
433  // See Case 1 above: Add a hole for the range we just moved.
434  InitializeIndex(fill_range.start(), fill_range.Length());
435  unused_indices_[0] = fill_range.start();
436 
437  // If we didn't completely fill unused_range, add a hole for the rest.
438  const IndexRange unused_hole(unused_range.start() + fill_range.Length(),
439  unused_range.end());
440  if (unused_hole.Length() > 0) {
441  InitializeIndex(unused_hole.start(), unused_hole.Length());
442  unused_indices_.push_back(unused_hole.start());
443  }
444  } else {
445  // See Case 2 above: Add a hole at the end of the range we shifted over.
446  const IndexRange shift_hole(unused_range.start() + fill_range.Length(),
447  fill_range.end());
448  InitializeIndex(shift_hole.start(), shift_hole.Length());
449  unused_indices_[0] = shift_hole.start();
450  }
451 
452  VerifyInternalState();
453  }

455  IndexRange LastIndexRangeSmallerThanHole(Index index) const {
456  // We want the last consecutive range of indices of length <= count.
457  const Count count = CountForIndex(index);
458 
459  // Loop from the back. `end` is the end of the range.
460  assert(unused_indices_.size() > 0);
461  size_t unused_i = unused_indices_.size() - 1;
462  for (Index end = num_indices(); end > index; end = PrevIndex(end)) {
463  // Skip over unused indices.
464  const Index unused_start = unused_indices_[unused_i];
465  const Index unused_end = NextIndex(unused_start);
466  assert(unused_end <= end);
467  if (end == unused_end) {
468  unused_i--;
469  continue;
470  }
471 
472  // Loop towards the front while the size still fits into `count`.
473  Index start = end;
474  for (Index j = PrevIndex(end); j > index; j = PrevIndex(j)) {
475  if (end - j > count) break;
476  if (j == unused_start) break;
477  start = j;
478  }
479 
480  // If at least some indices are in range, use those.
481  if (start < end) return IndexRange(start, end);
482  }
483 
484  // No index range fits, so return an invalid range.
485  return IndexRange();
486  }
487 
488  // When indices are moved or the number of inidices changes, we notify the
489  // caller via these callbacks.
490  CallbackInterface* callbacks_;
491 
492  // For every valid index, the number of indices
493  // associated with that index. For intermediate indices, negative number
494  // representing the offset to the actual index.
495  //
496  // valid indices
497  // | | | | |
498  // v v v v v
499  // For example: 1 | 2 -1 | 4 -1 -2 -3 | 1 | 1
500  // ^ ^ ^ ^
501  // | | | |
502  // offset to the actual index
503  std::vector<Count> counts_;
504 
505  // When an index is freed, we keep track of it here. When an index is
506  // allocated, we use one off this array, if one exists.
507  // When Defragment() is called, we empty this array by filling all the
508  // unused indices with the highest allocated indices. This reduces the total
509  // size of the data arrays.
510  std::vector<Index> unused_indices_;
511 };
